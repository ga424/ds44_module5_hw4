# C4 Documentation - Hadoop WordCount Application

This document provides C4 architecture diagrams for the Java WordCount MapReduce application and its local validation tooling.

## Level 1: System Context

```mermaid
C4Context
  title System Context - Hadoop WordCount Application

  Person(student, "Student/Operator", "Builds and runs WordCount job")
  System(wordcount, "WordCount Application", "Java MapReduce program packaged as a Maven-built JAR")
  System_Ext(hdfs, "HDFS", "Distributed storage for input and output data")
  System_Ext(yarn, "YARN", "Cluster resource manager and job scheduler")

  Rel(student, wordcount, "Builds, submits, and monitors job")
  Rel(wordcount, hdfs, "Reads input and writes output")
  Rel(wordcount, yarn, "Submits and executes MapReduce job")
```

## Level 3: Component Diagram

```mermaid
C4Component
  title Component Diagram - WordCount JAR

  Person(student, "Student/Operator", "Runs hadoop jar command")

  Container_Boundary(c1, "wordcount-hadoop-1.0.0.jar") {
    Component(driver, "WordCount main", "Java", "Configures and launches Hadoop Job")
    Component(strategy, "TextNormalizationStrategy", "Java", "Allows the mapper to swap text normalization rules")
    Component(mapper, "TokenizerMapper", "Java", "Normalizes input, tokenizes lines, and emits (word,1)")
    Component(jobFactory, "WordCountJobFactory", "Java", "Creates configured Hadoop Job instances")
    Component(combiner, "IntSumReducer as Combiner", "Java", "Locally aggregates mapper output")
    Component(reducer, "IntSumReducer as Reducer", "Java", "Aggregates final word counts")
  }

  Component_Ext(hdfsInput, "HDFS Input Split", "HDFS", "Input data blocks for map tasks")
  Component_Ext(hdfsOutput, "HDFS Output", "HDFS", "Final part files")

  Rel(student, driver, "Invokes")
  Rel(driver, jobFactory, "Uses")
  Rel(driver, strategy, "Uses")
  Rel(driver, mapper, "Sets mapper class")
  Rel(driver, combiner, "Sets combiner class")
  Rel(driver, reducer, "Sets reducer class")
  Rel(strategy, mapper, "Normalizes text")
  Rel(mapper, combiner, "Emits intermediate key-value pairs")
  Rel(combiner, reducer, "Sends aggregated intermediate data")
  Rel(hdfsInput, mapper, "Provides text records")
  Rel(reducer, hdfsOutput, "Writes final counts")
```

## Level 4: Deployment Diagram

```mermaid
C4Deployment
  title Deployment Diagram - 4-Node Hadoop Cluster

  Deployment_Node(cluster, "4-Node Hadoop Cluster", "Hadoop runtime", "One NameNode and three worker nodes") {

    Deployment_Node(master, "NameNode", "Linux VM", "HDFS NameNode, YARN ResourceManager, client tools") {
      Container(cli, "Hadoop Client Shell", "Bash", "Runs job commands from the cluster edge node")
      Container(jar, "wordcount-hadoop-1.0.0.jar", "JAR", "Packaged WordCount application")
    }

    Deployment_Node(slave1, "slave1", "Linux VM", "DataNode + NodeManager") {
      Container(mapTask1, "Map Task Executor", "JVM", "Executes map tasks")
      Container(reduceTask1, "Reduce Task Executor", "JVM", "Executes reduce tasks")
    }

    Deployment_Node(slave2, "slave2", "Linux VM", "DataNode + NodeManager") {
      Container(mapTask2, "Map Task Executor", "JVM", "Executes map tasks")
      Container(reduceTask2, "Reduce Task Executor", "JVM", "Executes reduce tasks")
    }

    Deployment_Node(slave3, "slave3", "Linux VM", "DataNode + NodeManager") {
      Container(mapTask3, "Map Task Executor", "JVM", "Executes map tasks")
      Container(reduceTask3, "Reduce Task Executor", "JVM", "Executes reduce tasks")
    }
  }

  Deployment_Node(devbox, "Developer Workstation", "macOS / Linux", "Runs Maven Wrapper, benchmark script, and local containerized validation") {
    Container(mvnw, "mvnw", "Bash", "Builds, tests, verifies coverage, and packages the JAR")
    Container(bench, "run_benchmarks.sh", "Bash", "Launches JMH benchmarks for WordCount hot paths")
    Container(localTest, "run_local_containerized.sh", "Bash", "Executes the Docker Compose Hadoop smoke test")
  }

  Deployment_Node(hdfs, "HDFS", "Distributed Storage", "Input and output files across DataNodes") {
    Container(hdfsSvc, "HDFS Services", "NameNode/DataNode", "Distributed file storage services")
  }

  Deployment_Node(yarn, "YARN", "Cluster Services", "ResourceManager and NodeManagers") {
    Container(yarnSvc, "YARN Services", "ResourceManager/NodeManager", "Cluster scheduling and container management")
  }

  Rel(devbox, mvnw, "Runs")
  Rel(devbox, bench, "Runs")
  Rel(devbox, localTest, "Runs")
  Rel(cli, jar, "Launches")
  Rel(jar, yarn, "Submits application")
  Rel(yarn, mapTask1, "Schedules containers")
  Rel(yarn, mapTask2, "Schedules containers")
  Rel(yarn, mapTask3, "Schedules containers")
  Rel(mapTask1, hdfs, "Reads input blocks")
  Rel(mapTask2, hdfs, "Reads input blocks")
  Rel(mapTask3, hdfs, "Reads input blocks")
  Rel(reduceTask1, hdfs, "Writes output")
  Rel(reduceTask2, hdfs, "Writes output")
  Rel(reduceTask3, hdfs, "Writes output")
```
